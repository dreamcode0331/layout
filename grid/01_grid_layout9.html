@supports (display: grid) {
 .grid_container {  /* do anything for grid supporting browsers here. */   }
} .grid_container ul {
display: grid; grid-template-columns: repeat(3, 1fr); gap: 20px; margin: 0; }
.grid_container li {
width: auto; min-height: auto; margin: 0; }
